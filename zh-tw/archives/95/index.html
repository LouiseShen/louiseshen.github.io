利用swift來實現自己的靜態部落格吧

  • 釋出時間：2020-07-20 12:38:20
  • 4162字數
  • 分類：
  1. 首頁
  2. 正文

寫在最前

之前把自己的Typecho部落格靜態化了，從最初的指令碼到獨立的私有的Typecho-Handsome-Java程式、從最初通過curl將html儲存下來到用JavaFX配合jdbc-sqlite ,han-lp等第三方庫完全自主將markdown渲染成html。

我從這個專案上得到了許多關於寶貴的經驗、歡樂與成就感，當然也付出了多個休息日。

不過Typecho-Handsome-Java (簡稱THJ)有兩個問題：

  • 從名字就可以看出它用到了Hamdsome主題，這也是為何私有的原因，而T則表示它用到了Typecho的資料庫表的命名（僅此而已）
  • JavaFX 8在macOS Catalina上有一個BUG，每次需要切換視窗再切換回來，否則選單欄無法使用

考慮到這兩個問題，以及因為我想學習swift開發還有各種各樣的原因，所以我建立了一個全新的Macondo程式，並且在github上以MIT License開源，當然完成這個專案需要移植整個THJ，並且需要一個全新或者仿照Handsome模樣（僅模樣）的預設主題。

完成這個專案需要時間，以及解決下面的問題

  • 適應swift的語法，以及中文swiftUI資料的匱乏
  • 一個sqlite的swift庫（已解決）
  • 一個渲染markdown的swift庫（已解決）

另外，我之前沒有學習過swift，更沒有swift開發經驗，簡而言之我就是swift萌新，所以如果有地方寫的不好的歡迎告知我修改。

平臺及工具選擇

  • macOS 10.15 Catalina（等macOS 11 Big Sur正式版釋出，也會升級）
  • Xcode 11.3（隨版本升級而升級）
  • 使用swiftUI

在Data.swift中新增用於文章和頁面的結構PostData

 import Foundation

 struct PostData{
     var cid : Int
     //'0' stands for 'post' while
     //'1' stands for 'page'.
     var type : Int
     //The 'slug' field has been deprecated, but you still need to let it has the same value of 'title'
     var title : String
     var slug : String
     var created : String
     var modified : String
     var text : String
     var thumbUrl : String
     var summary : String

     init() {
         cid = 0
         type = 0
         title = ""
         slug = ""
         created = ""
         modified = ""
         text = ""
         thumbUrl = ""
         summary = ""
     }

     init(cid : Int,type : Int,title : String,slug : String,created : String,modified : String,text : String,thumbUrl : String,summary : String) {
         self.cid = cid
         self.type = type
         self.title = title
         self.slug = title //let it has the same value of 'title'
         self.created = created
         self.modified = modified
         self.text = text
         self.thumbUrl = thumbUrl
         self.summary = summary
     }
 }

試用sqlite.swift來建立postDatas表

import Foundation
import SQLite

struct Sqlite {
    static func createTables(){
        var database : Connection
        let path = NSSearchPathForDirectoriesInDomains(.applicationSupportDirectory, .userDomainMask, true).first! + "/" + Bundle.main.bundleIdentifier!
        print(path)
        do{
            //create parent directory iff it doesn’t exist
            try FileManager.default.createDirectory(atPath: path, withIntermediateDirectories: true, attributes: nil)

            //open database
            database = try Connection("\(path)/blog.db")
        
            //postDatas table
            let pd = Table("postDatas")
            
            //postDatas columns settings
            let cid = Expression<Int64>("cid")
            let pdType = Expression<Int64>("type")
            let pdTitle = Expression<String>("title")
            let pdSlug = Expression<String>("slug")
            let pdCreated = Expression<String>("created")
            let pdModified = Expression<String>("modified")
            let pdText = Expression<String>("text")
            let pdThumbUrl = Expression<String>("thumbUrl")
            let pdSummary = Expression<String>("summary")
        
            //create postDatas table
            try database.run(pd.create{t in
                t.column(cid,primaryKey: true)
                t.column(pdType)
                t.column(pdTitle)
                t.column(pdSlug)
                t.column(pdCreated)
                t.column(pdModified)
                t.column(pdText)
                t.column(pdThumbUrl)
                t.column(pdSummary)
            })
        }catch{
            print(error)
        }
    }
}
